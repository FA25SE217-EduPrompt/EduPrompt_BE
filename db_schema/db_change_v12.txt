ALTER TABLE subscription_tiers
    ADD COLUMN points_limit BIGINT DEFAULT 0;

ALTER TABLE user_quota
    ADD COLUMN points_remaining BIGINT DEFAULT 0;

CREATE TABLE point_transactions (
                                    id UUID PRIMARY KEY,
                                    sender_id UUID NOT NULL REFERENCES users(id),
                                    receiver_id UUID NOT NULL REFERENCES users(id),
                                    amount BIGINT NOT NULL,
                                    status VARCHAR(50) NOT NULL DEFAULT 'completed',
                                    note TEXT,
                                    prompt_id UUID REFERENCES prompts(id),
                                    created_at TIMESTAMP WITH TIME ZONE DEFAULT now()
);

CREATE INDEX idx_point_transactions_sender_id ON point_transactions(sender_id);
CREATE INDEX idx_point_transactions_receiver_id ON point_transactions(receiver_id);
CREATE INDEX idx_point_transactions_prompt_id ON point_transactions(prompt_id);
